async function lancerCalcul() { markersLayer.clearLayers(); routeLayer.clearLayers(); segmentsTime = []; // Initialisation propre des compteurs let cumulDist = 0; let cumulDuree = 0; const mode = document.getElementById('modeSelect').value; const statsDiv = document.getElementById('stats'); let pointsBruts = allData.filter(d => zonesActivees.includes(d.Zone)).filter(c => c.Latitude && c.Longitude); if (pointsBruts.length === 0) { statsDiv.innerHTML = "Aucune donnée trouvée."; return; } pointsDeLaTournee = ordonnerPointsTSP(pointDepartCoords, pointsBruts); // Affichage des marqueurs (Départ + Commerces) L.marker(pointDepartCoords, { icon: L.divIcon({ className: 'numbered-marker start-marker', html: `🏠`, iconSize: [26, 26] }) }).addTo(markersLayer); pointsDeLaTournee.forEach((p, i) => { const nom = p["NOM DANS LISTING"] || "Commerce"; const icon = L.divIcon({ className: `numbered-marker marker-id-${i}`, html: `${i+1}`, iconSize: [22, 22] }); L.marker([p.Latitude, p.Longitude], { icon: icon }) .addTo(markersLayer) .bindPopup(`${nom}
${p.ADRESSE}`) .bindTooltip(`${i+1}. ${nom}`, { direction: 'top', className: 'leaflet-tooltip-own' }); }); let allCoords = [pointDepartCoords[1] + "," + pointDepartCoords[0]]; pointsDeLaTournee.forEach(p => allCoords.push(p.Longitude + "," + p.Latitude)); try { statsDiv.innerHTML = "⌛ Calcul des trajets en cours..."; const chunkSize = 15; for (let i = 0; i < allCoords.length - 1; i += chunkSize - 1) { const chunk = allCoords.slice(i, i + chunkSize); const url = `https://router.project-osrm.org/route/v1/${mode}/${chunk.join(';')}?overview=full&geometries=geojson`; const r = await fetch(url); const d = await r.json(); if (d.code === 'Ok') { const routePart = d.routes[0]; L.geoJSON(routePart.geometry, { style: { color: '#2563eb', weight: 4, opacity: 0.4 } }).addTo(routeLayer); // Somme des données du tronçon cumulDist += parseFloat(routePart.distance); cumulDuree += parseFloat(routePart.duration); routePart.legs.forEach(leg => { segmentsTime.push(Math.round(leg.duration / 60)); }); } // Petit délai pour éviter l'erreur 429 await new Promise(res => setTimeout(res, 150)); } // AFFICHAGE FINAL (Une fois la boucle terminée) statsDiv.innerHTML = ` Résumé de la zone :
📏 Distance totale : ${(cumulDist / 1000).toFixed(1)} km
⏱️ Temps total : ${Math.round(cumulDuree / 60)} min
📍 Arrêts : ${pointsDeLaTournee.length} `; document.getElementById('nav-panel').style.display = 'block'; majUIEtape(); map.fitBounds(markersLayer.getBounds(), { padding: [40, 40] }); } catch (e) { console.error(e); statsDiv.innerHTML = "⚠️ Erreur lors du calcul des statistiques."; } }

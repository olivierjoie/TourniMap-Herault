async function lancerCalcul() { markersLayer.clearLayers(); routeLayer.clearLayers(); segmentsTime = []; const mode = document.getElementById('modeSelect').value; let pointsBruts = allData.filter(d => zonesActivees.includes(d.Zone)).filter(c => c.Latitude && c.Longitude); if (pointsBruts.length === 0) return; pointsDeLaTournee = ordonnerPointsTSP(pointDepartCoords, pointsBruts); // Marqueurs L.marker(pointDepartCoords, { icon: L.divIcon({ className: 'numbered-marker start-marker', html: `🏠`, iconSize: [26, 26] }) }).addTo(markersLayer); pointsDeLaTournee.forEach((p, i) => { const icon = L.divIcon({ className: `numbered-marker marker-id-${i}`, html: `${i+1}`, iconSize: [22, 22] }); L.marker([p.Latitude, p.Longitude], { icon: icon }).addTo(markersLayer).bindPopup(`${p["NOM DANS LISTING"]}`); }); // Préparation des coordonnées let allCoords = [pointDepartCoords[1] + "," + pointDepartCoords[0]]; pointsDeLaTournee.forEach(p => allCoords.push(p.Longitude + "," + p.Latitude)); let totalDist = 0; let totalDuree = 0; try { document.getElementById('stats').innerHTML = "⌛ Calcul des tronçons..."; // DÉCOUPAGE EN TRONÇONS DE 15 POINTS (Évite l'erreur 400 et 429) const chunkSize = 15; for (let i = 0; i < allCoords.length - 1; i += chunkSize - 1) { const chunk = allCoords.slice(i, i + chunkSize); const url = `https://router.project-osrm.org/route/v1/${mode}/${chunk.join(';')}?overview=full&geometries=geojson`; const r = await fetch(url); const d = await r.json(); if (d.code === 'Ok') { const routePart = d.routes[0]; L.geoJSON(routePart.geometry, { style: { color: '#2563eb', weight: 4, opacity: 0.4 } }).addTo(routeLayer); totalDist += routePart.distance; totalDuree += routePart.duration; routePart.legs.forEach(leg => { segmentsTime.push(Math.round(leg.duration / 60)); }); } // Petit délai de 100ms pour ne pas brusquer le serveur await new Promise(resolve => setTimeout(resolve, 100)); } document.getElementById('stats').innerHTML = ` Résumé Tournée :
📏 Distance : ${(totalDist / 1000).toFixed(1)} km
⏱️ Temps : ${Math.round(totalDuree / 60)} min
📍 Arrêts : ${pointsDeLaTournee.length} `; document.getElementById('nav-panel').style.display = 'block'; majUIEtape(); map.fitBounds(markersLayer.getBounds(), { padding: [40, 40] }); } catch (e) { console.error(e); document.getElementById('stats').innerHTML = "⚠️ Erreur de calcul (Vérifiez votre connexion)."; } }

async function lancerCalcul() { markersLayer.clearLayers(); routeLayer.clearLayers(); segmentsTime = []; const mode = document.getElementById('modeSelect').value; // Filtrage des points de la zone sélectionnée let pointsBruts = allData.filter(d => zonesActivees.includes(d.Zone)).filter(c => c.Latitude && c.Longitude); if (pointsBruts.length === 0) return; // Calcul de l'ordre le plus court (TSP) pointsDeLaTournee = ordonnerPointsTSP(pointDepartCoords, pointsBruts); // Ajout visuel du départ L.marker(pointDepartCoords, { icon: L.divIcon({ className: 'numbered-marker start-marker', html: `🏠`, iconSize: [26, 26] }) }).addTo(markersLayer); // Ajout des points de livraison numérotés pointsDeLaTournee.forEach((p, i) => { const icon = L.divIcon({ className: `numbered-marker marker-id-${i}`, html: `${i+1}`, iconSize: [22, 22] }); L.marker([p.Latitude, p.Longitude], { icon: icon }).addTo(markersLayer).bindPopup(`${p["NOM DANS LISTING"]}`); }); // Préparation de la liste des coordonnées pour l'appel UNIQUE let coords = [pointDepartCoords[1] + "," + pointDepartCoords[0]]; pointsDeLaTournee.forEach(p => coords.push(p.Longitude + "," + p.Latitude)); try { document.getElementById('stats').innerHTML = "⌛ Calcul de l'itinéraire..."; // APPEL UNIQUE : On envoie tous les points d'un coup (Évite le blocage 429) const fullUrl = `https://router.project-osrm.org/route/v1/${mode}/${coords.join(';')}?overview=full&geometries=geojson`; const r = await fetch(fullUrl); const d = await r.json(); if (d.code === 'Ok') { const routeObj = d.routes[0]; // Tracé de la ligne sur la carte L.geoJSON(routeObj.geometry, { style: { color: '#2563eb', weight: 4, opacity: 0.4 } }).addTo(routeLayer); // Calcul des stats globales depuis la réponse unique const totalDist = (routeObj.distance / 1000).toFixed(1); // km const totalDuree = Math.round(routeObj.duration / 60); // min // Extraction des temps de chaque segment (legs) pour la navigation routeObj.legs.forEach(leg => { segmentsTime.push(Math.round(leg.duration / 60)); }); // Affichage des statistiques dans la barre latérale document.getElementById('stats').innerHTML = ` Résumé Tournée :
📏 Distance : ${totalDist} km
⏱️ Temps : ${totalDuree} min
📍 Arrêts : ${pointsDeLaTournee.length} `; document.getElementById('nav-panel').style.display = 'block'; majUIEtape(); map.fitBounds(markersLayer.getBounds(), { padding: [40, 40] }); } } catch (e) { console.error(e); document.getElementById('stats').innerHTML = "⚠️ Erreur serveur (Trop de points ou zone inaccessible)."; } }

async function lancerCalcul() { markersLayer.clearLayers(); routeLayer.clearLayers(); segmentsTime = []; // On vide les temps par segment const mode = document.getElementById('modeSelect').value; // Filtrage des points de la zone let pointsBruts = allData.filter(d => zonesActivees.includes(d.Zone)).filter(c => c.Latitude && c.Longitude); if (pointsBruts.length === 0) return; // Calcul de l'ordre (TSP) pointsDeLaTournee = ordonnerPointsTSP(pointDepartCoords, pointsBruts); // Affichage du point de départ L.marker(pointDepartCoords, { icon: L.divIcon({ className: 'numbered-marker start-marker', html: `🏠`, iconSize: [26, 26] }) }).addTo(markersLayer); // Affichage des points de livraison pointsDeLaTournee.forEach((p, i) => { const icon = L.divIcon({ className: `numbered-marker marker-id-${i}`, html: `${i+1}`, iconSize: [22, 22] }); L.marker([p.Latitude, p.Longitude], { icon: icon }).addTo(markersLayer).bindPopup(`${p["NOM DANS LISTING"]}`); }); // Préparation des coordonnées pour l'API unique let coords = [pointDepartCoords[1] + "," + pointDepartCoords[0]]; pointsDeLaTournee.forEach(p => coords.push(p.Longitude + "," + p.Latitude)); try { document.getElementById('stats').innerHTML = "⌛ Calcul de l'itinéraire global..."; // UNE SEULE REQUÊTE pour toute la zone (évite l'erreur 429) const fullUrl = `https://router.project-osrm.org/route/v1/${mode}/${coords.join(';')}?overview=full&geometries=geojson`; const r = await fetch(fullUrl); const d = await r.json(); if (d.code === 'Ok') { const routeGlobale = d.routes[0]; // Affichage du tracé bleu L.geoJSON(routeGlobale.geometry, { style: { color: '#2563eb', weight: 4, opacity: 0.3 } }).addTo(routeLayer); // --- RÉCUPÉRATION DES STATS GLOBALES --- const totalDist = routeGlobale.distance; // en mètres const totalDuree = routeGlobale.duration; // en secondes // On remplit segmentsTime pour la navigation point par point routeGlobale.legs.forEach(leg => { segmentsTime.push(Math.round(leg.duration / 60)); // conversion en minutes }); // Mise à jour de l'affichage Stats document.getElementById('stats').innerHTML = ` Résumé de la tournée :
📏 Distance totale : ${(totalDist/1000).toFixed(1)} km
⏱️ Temps de trajet : ${Math.round(totalDuree/60)} min
📍 Nombre d'arrêts : ${pointsDeLaTournee.length} `; document.getElementById('nav-panel').style.display = 'block'; majUIEtape(); map.fitBounds(markersLayer.getBounds(), { padding: [40, 40] }); } } catch (e) { console.error(e); document.getElementById('stats').innerHTML = "⚠️ Erreur lors du calcul global."; } }

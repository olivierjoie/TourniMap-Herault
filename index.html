async function lancerCalcul() { markersLayer.clearLayers(); routeLayer.clearLayers(); segmentsTime = []; let cumulDist = 0; // On initialise les compteurs à 0 let cumulDuree = 0; const mode = document.getElementById('modeSelect').value; let pointsBruts = allData.filter(d => zonesActivees.includes(d.Zone)).filter(c => c.Latitude && c.Longitude); if (pointsBruts.length === 0) return; pointsDeLaTournee = ordonnerPointsTSP(pointDepartCoords, pointsBruts); // Ajout des marqueurs (Départ + Commerces) L.marker(pointDepartCoords, { icon: L.divIcon({ className: 'numbered-marker start-marker', html: `🏠`, iconSize: [26, 26] }) }).addTo(markersLayer); pointsDeLaTournee.forEach((p, i) => { const nom = p["NOM DANS LISTING"] || "Commerce"; const icon = L.divIcon({ className: `numbered-marker marker-id-${i}`, html: `${i+1}`, iconSize: [22, 22] }); L.marker([p.Latitude, p.Longitude], { icon: icon }) .addTo(markersLayer) .bindPopup(`${nom}
${p.ADRESSE}`) .bindTooltip(`${i+1}. ${nom}`, { direction: 'top', className: 'leaflet-tooltip-own' }); }); let allCoords = [pointDepartCoords[1] + "," + pointDepartCoords[0]]; pointsDeLaTournee.forEach(p => allCoords.push(p.Longitude + "," + p.Latitude)); try { document.getElementById('stats').innerHTML = "⌛ Calcul des trajets..."; const chunkSize = 15; for (let i = 0; i < allCoords.length - 1; i += chunkSize - 1) { const chunk = allCoords.slice(i, i + chunkSize); const url = `https://router.project-osrm.org/route/v1/${mode}/${chunk.join(';')}?overview=full&geometries=geojson`; const r = await fetch(url); const d = await r.json(); if (d.code === 'Ok') { const routePart = d.routes[0]; L.geoJSON(routePart.geometry, { style: { color: '#2563eb', weight: 4, opacity: 0.4 } }).addTo(routeLayer); // --- SOMME DES DONNÉES --- cumulDist += routePart.distance; cumulDuree += routePart.duration; routePart.legs.forEach(leg => { segmentsTime.push(Math.round(leg.duration / 60)); }); } await new Promise(res => setTimeout(res, 150)); } // --- AFFICHAGE FINAL BASÉ SUR LA SOMME --- document.getElementById('stats').innerHTML = ` Résumé de la zone :
📏 Distance totale : ${(cumulDist / 1000).toFixed(1)} km
⏱️ Temps total estimé : ${Math.round(cumulDuree / 60)} min
📍 Nombre d'arrêts : ${pointsDeLaTournee.length} `; document.getElementById('nav-panel').style.display = 'block'; majUIEtape(); map.fitBounds(markersLayer.getBounds(), { padding: [40, 40] }); } catch (e) { console.error(e); document.getElementById('stats').innerHTML = "⚠️ Erreur lors du calcul des segments."; } }
